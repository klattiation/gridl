export default function(context, stateProvider) {

    const { columns, rows } = stateProvider.getState();

    /**
     * Get the current size of the grid.
     *
     * @memberOf gridl
     * @method
     * @instance
     *
     * @returns {number[]}
     */
    function size() {
        return [columns, rows];
    }

    return size;
}
